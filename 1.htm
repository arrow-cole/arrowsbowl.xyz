import { Resend } from "resend"; // Resend integration - managed via Replit connector let connectionSettings: any; async function getCredentials() { const hostname = process.env.REPLIT_CONNECTORS_HOSTNAME; const xReplitToken = process.env.REPL_IDENTITY ? "repl " + process.env.REPL_IDENTITY : process.env.WEB_REPL_RENEWAL ? "depl " + process.env.WEB_REPL_RENEWAL : null; if (!xReplitToken) { throw new Error("X_REPLIT_TOKEN not found for repl/depl"); } connectionSettings = await fetch( "https://" + hostname + "/api/v2/connection?include_secrets=true&connector_names=resend", { headers: { Accept: "application/json", X_REPLIT_TOKEN: xReplitToken, }, }, ) .then((res) => res.json()) .then((data) => data.items?.[0]); if (!connectionSettings || !connectionSettings.settings.api_key) { throw new Error("Resend not connected"); } return { apiKey: connectionSettings.settings.api_key, fromEmail: connectionSettings.settings.from_email, }; } async function getUncachableResendClient() { const { apiKey, fromEmail } = await getCredentials(); return { client: new Resend(apiKey), fromEmail, }; } const ADMIN_EMAIL = "forms@mibowling.us"; export async function sendHostRequestNotification( username: string, displayName: string | null, email: string | null, ) { try { const { client, fromEmail } = await getUncachableResendClient(); const applicantName = displayName || username; const applicantEmail = email ? ` (${email})` : ""; const senderEmail = "noreply@mibowling.us"; console.log( `Sending host request email from: ${senderEmail} to: ${ADMIN_EMAIL}`, ); const result = await client.emails.send({ from: senderEmail, to: ADMIN_EMAIL, subject: `New Host Access Request - ${applicantName}`, html: `

MI Bowling Tournaments

New Host Access Request

A user has requested host access on MI Bowling Tournaments.

${ email ? `` : "" }
Username ${username}
Display Name ${applicantName}
Email ${email}

Log in to the Admin Panel to approve or deny this request.

This is an automated notification from MI Bowling Tournaments.

`, }); console.log(`Host request email result:`, JSON.stringify(result)); } catch (error) { console.error("Failed to send host request notification email:", error); } }
